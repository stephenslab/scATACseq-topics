Perform PCA on the topic proportions, and use these PCs to define clusters.

pca <- prcomp(poisson2multinom(fit)$L)$x
x   <- rep("A",nrow(pca))
pc1 <- pca[,1]
pc2 <- pca[,2]
pc3 <- pca[,3]
pc4 <- pca[,4]
pc5 <- pca[,5]
pc6 <- pca[,6]
pc8 <- pca[,8]
x[pc1 < 0.1]  <- "B"
x[pc2 < -0.2] <- "C"
x[pc3 > 0.35] <- "pDC"
x[pc3 < -0.1 & pc4 < 0.05] <- "GMP"
x[pc4 > 0.55] <- "CLP"
x[pc5 < -0.25 & pc6 < -0.25] <- "mono"
x[pc8 > 0.35] <- "U"
samples$cluster <- factor(x)
